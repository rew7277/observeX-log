Code Examples

curl -X POST {{ request.host_url }}api/v1/logs/ingest \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "environment": "PROD",
    "application": "payment-engine",
    "logs": "2024-01-15 12:00:01 ERROR timeout\n2024-01-15 12:00:02 INFO ok"
  }'
import requests

response = requests.post(
    "{{ request.host_url }}api/v1/logs/ingest",
    headers={"Authorization": "Bearer YOUR_API_KEY"},
    json={
        "environment": "PROD",
        "application": "payment-engine",
        "logs": open("app.log").read()
    }
)
print(response.json())
const fs = require('fs');
const res = await fetch('{{ request.host_url }}api/v1/logs/ingest', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    environment: 'PROD',
    application: 'payment-engine',
    logs: fs.readFileSync('app.log', 'utf8')
  })
});
console.log(await res.json());
